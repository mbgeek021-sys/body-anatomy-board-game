<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Body Anatomy Board Game</title>
  </head>
  <body style="margin:0;background:#0f172a;color:white;font-family:sans-serif;">
    <div id="app"></div>

    <script>
      const app = document.getElementById("app");

      let playerPos = 0;

      const board = Array.from({ length: 30 }, (_, i) => i);

      function rollDice() {
        const roll = Math.floor(Math.random() * 6) + 1;
        playerPos += roll;
        if (playerPos >= board.length - 1) {
          alert("You reached the brain 🧠 YOU WIN!");
          playerPos = 0;
        }
        render();
      }

      function render() {
        app.innerHTML = `
          <div style="display:flex;height:100vh;">
            
            <div style="flex:1;display:flex;align-items:center;justify-content:center;">
              <div style="display:grid;grid-template-columns:repeat(6,80px);gap:10px;">
                ${board.map((i) => `
                  <div style="
                    height:80px;
                    border-radius:12px;
                    background:${i === playerPos ? '#22c55e' : '#1e293b'};
                    display:flex;
                    align-items:center;
                    justify-content:center;
                    font-size:12px;
                  ">
                    ${i === 0 ? "Start" : i === board.length-1 ? "Brain 🧠" : "Body"}
                  </div>
                `).join("")}
              </div>
            </div>

            <div style="width:300px;padding:20px;background:#020617;">
              <h2>Controls</h2>
              <button onclick="rollDice()" style="
                width:100%;
                padding:15px;
                border:none;
                border-radius:10px;
                background:#14b8a6;
                color:white;
                font-size:16px;
                cursor:pointer;
              ">
                🎲 Roll Dice
              </button>
              <p style="margin-top:20px;">Position: ${playerPos}</p>
            </div>

          </div>
        `;
      }

      render();
    </script>
  </body>
</html>
